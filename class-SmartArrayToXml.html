Class SmartArrayToXml

Class: SmartArrayToXml - Create XML from a PHP Array.

//-- Sample use:
$array = array(
      'xml' => array(
              'id' => '101',
              'name' => 'Test',
              'files' => array(
                      'file1' => '01.txt',
                      'file2' => '02.txt',
                      'file3' => '03.txt'
              ),
              'date' => '2014-12-15 09:30:00'
      )
);
$obj = new SmartArrayToXml();
$xml = $obj->compose_xml($array);
//--
